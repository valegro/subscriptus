 88:     def self.conditions_to_criteria(conditions, camelize_field = false, criterion_order = 1)
 89:       criteria = []
 90:       return criteria if conditions.blank?
 91: 
 92:       conditions.each_pair { |k,v|
 93:         criterion = CmCriterion.new
 94:         match = k.to_s.match(/([\w{}\d]+)([<>=!]+)?/)
 95:         criterion.fieldName = match[1]
 96:         criterion.fieldName = criterion.fieldName.camelize if camelize_field
 97:         criterion.operator = Operators[ ( match[2] || '==' ) ]
 98:         criterion.operand = v
 99: 
100:         # soap4r complains without these
101:         criterion.order = criterion_order
102:         criterion.logicalOperator = LogicalOperator::And
103:         criterion.leftParentheses = 1
104:         criterion.rightParentheses = 1
105: 
106:         criteria.push(criterion)
107:         criterion_order += 1
108:       }
109:       return criteria
110:     end

get_recipients(conditions)

you could specify operation in the form like: :’email_address!=’ => ‘foo’, or :’last_modified>=’ => ‘2010-09-01T12:00:00.000’ by default, this assumes operator is ’==’ Note :’last_modfied=>’ will be interpreted as :’last_modified==’ due to ’=>’ not being a valid operator.
